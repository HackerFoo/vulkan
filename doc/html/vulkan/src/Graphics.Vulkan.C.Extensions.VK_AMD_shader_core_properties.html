  -- number of SGPRs allocated for a wave.

  -- number of SGPRs allocated for a wave.
  vkMaxSgprAllocation :: Word32

  -- granularity of SGPR allocation for a wave.

  -- physical Vector General Purpose Registers (VGPRs) per SIMD.
